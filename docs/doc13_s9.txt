A gap of one day or more breaks the streak.  -- Current logging streak WITH ordered_dates AS (   SELECT log_date   FROM progress_logs   WHERE user_id = $userId   ORDER BY log_date DESC ), anchor AS (   -- Start from today if logged today, otherwise yesterday   SELECT     CASE       WHEN EXISTS (         SELECT 1 FROM progress_logs         WHERE user_id = $userId AND log_date = CURRENT_DATE       ) THEN CURRENT_DATE       ELSE CURRENT_DATE - INTERVAL &apos;1 day&apos;     END AS start_date ), streak_calc AS (   SELECT     log_date,     ROW_NUMBER() OVER (ORDER BY log_date DESC) AS rn,     (SELECT start_date FROM anchor)       - (ROW_NUMBER() OVER (ORDER BY log_date DESC) - 1) AS expected_date   FROM ordered_dates   WHERE log_date &lt;= (SELECT start_date FROM anchor) ) SELECT COUNT(*) AS current_streak FROM streak_calc WHERE log_date = expected_date  TypeScript implementation (for use alongside SQL) function getCurrentStreak(logDates: Date[]): number {   if (logDates.length === 0) return 0    const sorted = [...logDates].sort((a, b) =&gt; b.getTime() - a.getTime())   const today = new Date()   today.setHours(0, 0, 0, 0)   const yesterday = new Date(today)   yesterday.setDate(yesterday.getDate() - 1)    // If most recent log is older than yesterday, streak is 0   if (sorted[0].getTime() &lt; yesterday.getTime()) return 0    // Anchor: start from today if logged, else yesterday   const anchor = sorted[0].getTime() === today.getTime() ?
